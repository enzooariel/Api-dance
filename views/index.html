Authentication

Register User

URL: /api/auth/register
Method: POST
Headers: None
Body:
{
  "username": "testuser",
  "password": "testpassword"
}

Login User

URL: /api/auth/login
Method: POST
Headers: None
Body:
{
  "username": "testuser",
  "password": "testpassword"
}

Dance Styles

Get All Dance Styles

URL: /api/dance-styles
Method: GET
Headers:
  Authorization: Bearer <your_jwt_token>

Create New Dance Style

URL: /api/dance-styles
Method: POST
Headers:
  Authorization: Bearer <your_jwt_token>
Body:
{
  "name": "Salsa",
  "origin": "Cuba",
  "description": "A popular dance style originating from Latin America."
}

Update Dance Style

URL: /api/dance-styles/<id_of_dance_style>
Method: PUT
Headers:
  Authorization: Bearer <your_jwt_token>
Body:
{
  "name": "Salsa Updated",
  "origin": "Cuba",
  "description": "An updated description for Salsa."
}

Delete Dance Style

URL: /api/dance-styles/<id_of_dance_style>
Method: DELETE
Headers:
  Authorization: Bearer <your_jwt_token>

Locations

Get All Locations

URL: /api/locations
Method: GET
Headers:
  Authorization: Bearer <your_jwt_token>

Create New Location

URL: /api/locations
Method: POST
Headers:
  Authorization: Bearer <your_jwt_token>
Body:
{
  "name": "Dance Studio",
  "address": "123 Main St",
  "city": "Havana",
  "country": "Cuba"
}

Update Location

URL: /api/locations/<id_of_location>
Method: PUT
Headers:
  Authorization: Bearer <your_jwt_token>

  "name": "Dance Studio Updated",

  "country": "Cuba"
}

Delete Location

URL: /api/locations/<id_of_location>

Headers:
  Authorization: Bearer <your_jwt_token>
